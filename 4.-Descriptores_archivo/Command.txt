# TE CREAS UN ARCHIVO LLAMADO "file"
exec 3<> file
# ESTE ARCHIVO SE IDENTIFICA CON EL NÚMERO "3" TIENE CAPACIDAD DE LECTURA (<) Y ESCRITURA (>)


# ESTA ES UNA MANERA DE ENVIAR EL OUTPUT A ESE DESCRIPTOR DE ARCHIVO, ESTO LO ALMACENA EN FORMATO APPEND (AGREGAR)
whoami >&3


# LO CERRAMOS DE ESTA MANERA, NO PIERDE SU CONTENIDO, SOLO YA NO PUEDES ESCRIBIR (AGREGAR CONTENIDO) EN ÉL DE LA MANERA ANTERIOR
exec 3>&-


# CREAMOS OTRO LLAMADO DATA CON LA REFERENCIA DEL NÚMERO 5 CON CAPACIDAD DE ESCRITURA Y LECTURA
exec 5<> data

# CREAR UNA COPIA DE IDENTIFICADORES DE ARCHIVO (ESTAMOS CREANDO UNA COPIA DEL REFERENCIA 5) LA COPIA ES EL REFERENCIA 8
exec 8>&5 
# CUALQUIER COSA QUE META EN EL REFERENCIADO 8 ESTARA EN EL 5


# CERRAMOS EL REFERENCIADO 5 Y SOLO SE PODRA AGREGAR CONTENIDO AL REFERENCIADO 8
exec 5>&- 


# CREANDO COPIA Y CERRANDO  DEL 5 Y CERRANDO ESE MISMO
exec 5<> example
exec 6>&5-
